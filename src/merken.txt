import React, { useState } from 'react';
import Accordion from '@mui/material/Accordion';
import AccordionSummary from '@mui/material/AccordionSummary';
import AccordionDetails from '@mui/material/AccordionDetails';
import ExpandMoreIcon from '@mui/icons-material/ExpandMore';
import Typography from '@mui/material/Typography';

function MyAccordionComponent({ shouldResetOnRerender }) {
  // Initialer Zustand (z.B. false für geschlossen)
  const [expanded, setExpanded] = useState(false); 

  // Wenn du möchtest, dass es bei einem bestimmten Prop-Wechsel zurückgesetzt wird:
  // Du könntest ein useEffect verwenden, um den Zustand zurückzusetzen, 
  // wenn sich 'shouldResetOnRerender' ändert.
  // Hier ist ein einfacherer Weg: Das `key`-Prop.

  // ---

  // Handler für die Änderung des Zustands
  const handleChange = () => {
    setExpanded(!expanded);
  };

  return (
    <Accordion 
      expanded={expanded} // **Kontrolliert** den Expansionszustand
      onChange={handleChange} // **Aktualisiert** den Expansionszustand bei Klick
    >
      <AccordionSummary
        expandIcon={<ExpandMoreIcon />}
        aria-controls="panel1a-content"
        id="panel1a-header"
      >
        <Typography>Accordion Header</Typography>
      </AccordionSummary>
      <AccordionDetails>
        <Typography>
          Der Inhalt des Accordion.
        </Typography>
      </AccordionDetails>
    </Accordion>
  );
}

export default MyAccordionComponent;
